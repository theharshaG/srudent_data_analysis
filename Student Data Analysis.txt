#student data
students = {
    "Harsha": 85,
    "Ravi": 72,
    "Anu": 90,
    "Kiran": 60
}
# above 75 scored students
for name,mark in students.items():
    if mark>75:
        print(name,mark)
        
#maximum mark
max_value=float ('-inf')
for x in students.values():
    if x > max_value:
        max_value = x
print(f"The maximum marks is: {max_value}")

#class topper
topper = max(students, key=lambda x: students[x])
print("Topper:", topper)

#tp 3 students
sorted_students = sorted(students.items(), key=lambda x: x[1], reverse=True)
print("Top 3 Students:")
for name, marks in sorted_students[:3]:
    print(name, marks)
    
#final result pass/fail   
result = {name: ("Pass" if mark >= 75 else "Fail") 
          for name, mark in students.items()}

print(result)
